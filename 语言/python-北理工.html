def fact(n,*b):
    s = 1
    for i in range(1.n+1):
        s *= i
    for item in b:
        s *= item
    return s

def getHTMLText(url):

        r = requests.get(url,timeout=30)
        r.raise_for_status()
        r.encoding = r.apparent_encoding
        return r.text
    except:
        return "产生异常"
